```java
class Solution {
    public int lengthOfLongestSubstring(String s) {
        Set<Character> occ = new HashSet<Character>();
        int n = s.length();
        int rk = 0, ans = 0;
        for (int i = 0; i < n; ++i) {
            if (i != 0) {
                // 注意，这里容易错，是 remove (i - 1) 个字符
                occ.remove(s.charAt(i - 1));
            }
            while (rk < n && !occ.contains(s.charAt(rk))) {
                occ.add(s.charAt(rk));
                ++rk;
            }
            ans = Math.max(ans, rk - i );
        }
        return ans;
    }
}
```
