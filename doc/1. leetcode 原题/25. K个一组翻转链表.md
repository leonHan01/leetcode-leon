```java
public class Solution {
    public ListNode reverseKGroup(ListNode head, int k) {
        if (head == null || k == 1) {
            return head;
        }
        
        // 哑结点，作为返回链表的头部前缀
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        
        // 初始化指针
        ListNode cur = dummy, nxt = dummy, pre = dummy;
        int count = 0;
        
        // 遍历链表计算长度
        while (cur.next != null) {
            cur = cur.next;
            count++;
        }
        
        // 遍历并翻转每k个一组的链表部分
        while (count >= k) {
            cur = pre.next;  // 当前处理段的第一个节点
            nxt = cur.next;  // 当前处理段的第二个节点
            for (int i = 1; i < k; i++) {
                // 进行节点翻转
                cur.next = nxt.next;
                nxt.next = pre.next;
                pre.next = nxt;
                nxt = cur.next;
            }
            pre = cur;
            count -= k;
        }
        
        return dummy.next;  // 返回哑结点的下一个节点，即真正的头节点
    }
}
```

```java

class Solution {
    public ListNode reverseKGroup(ListNode head, int k) {
        ListNode hair = new ListNode(0);
        hair.next = head;
        ListNode pre = hair;

        while (head != null) {
            ListNode tail = pre;
            // 查看剩余部分长度是否大于等于 k
            for (int i = 0; i < k; ++i) {
                tail = tail.next;
                if (tail == null) {
                    return hair.next;
                }
            }
            ListNode nex = tail.next;
            ListNode[] reverse = myReverse(head, tail);
            head = reverse[0];
            tail = reverse[1];
            // 把子链表重新接回原链表
            pre.next = head;
            tail.next = nex;
            pre = tail;
            head = tail.next;
        }

        return hair.next;
    }

    public ListNode[] myReverse(ListNode head, ListNode tail) {
        ListNode prev = tail.next;
        ListNode p = head;
        while (prev != tail) {
            ListNode nex = p.next;
            p.next = prev;
            prev = p;
            p = nex;
        }
        return new ListNode[]{tail, head};
    }
}
```
